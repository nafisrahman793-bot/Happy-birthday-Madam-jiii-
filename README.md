    <script>
        // Function to reveal the surprise message
        function revealSurprise() {
            const surpriseDiv = document.getElementById('surprise');
            surpriseDiv.style.display = 'block';
            
            // Create a burst of extra hearts when clicked
            for(let i=0; i<15; i++) {
                setTimeout(createHeart, i * 100);
            }
        }

        // Function to generate floating background hearts
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart');
            heart.innerHTML = '❤️';
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.animationDuration = Math.random() * 3 + 3 + 's'; // 3s to 6s
            heart.style.fontSize = Math.random() * 20 + 15 + 'px';
            
            document.getElementById('hearts-container').appendChild(heart);

            // Remove heart after animation ends
            setTimeout(() => {
                heart.remove();
            }, 6000);
        }

        // Keep generating background hearts smoothly
        setInterval(createHeart, 400);
